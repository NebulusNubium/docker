CREATE DATABASE pokeworld

CREATE TABLE type (
    id INT AUTO_INCREMENT PRIMARY KEY,
    type VARCHAR(20)
);

CREATE TABLE pokemon (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nom VARCHAR(20)
);


CREATE TABLE dresseurs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nom VARCHAR(20),
    age INT,
    ville_id INT,
	pokemon_id INT,
    FOREIGN KEY (ville_id) REFERENCES villes(id),
    FOREIGN KEY (pokemon_id) REFERENCES pokemons(id)
);
